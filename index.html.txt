        function handleEditPassword(username) {
            if (state.currentRole !== 'admin') {
                showToast("Bạn không có quyền đổi mật khẩu!", "shield-alert");
                return;
            }

            const newPass = prompt("Nhập mật khẩu mới cho tài khoản '" + username + "':");
            if (newPass === null) return;
            
            const passTrimmed = newPass.trim();
            if (passTrimmed === '') {
                showToast("Mật khẩu không được để trống!", "alert-triangle");
                return;
            }

            const index = state.accounts.findIndex(acc => acc.username === username);
            if (index !== -1) {
                state.accounts[index].password = passTrimmed;
                saveToStorage();
                showToast("Đã đổi mật khẩu thành công!");
            }
        }

            textArea.value = text;
            textArea.style.top = "0";
            textArea.style.left = "0";
            textArea.style.position = "fixed";
            document.body.appendChild(textArea);

            } catch (err) {
                console.error('Không thể sao chép văn bản', err);
            }
            document.body.removeChild(textArea);
            return success;
        }

                <button type="button" onclick="handleSeatClick(${i})" class="relative py-2.5 rounded-lg border text-base transition-all duration-150 flex flex-col items-center justify-center ${btnClass}">
                    <span>${i}</span>
                    ${indicator}
                </button>
                `;
            }

            container.innerHTML = html;
        }

        function handleSeatClick(seatNum) {
            if (state.multiSelectMode) {
                if (state.selectedSeats.includes(seatNum)) {
                    if (state.selectedSeats.length > 1) {
                        state.selectedSeats = state.selectedSeats.filter(item => item !== seatNum);
                    } else {
                        showToast("Cần giữ lại ít nhất 1 ghế đang chọn!", "alert-triangle");
                    }
                } else {
                    state.selectedSeats.push(seatNum);
                    state.selectedSeats.sort((a, b) => a - b);
                }
            } else {
                state.selectedSeats = [seatNum];
            }
            
            saveToStorage();
            loadSeatsToForm();
        }

        // Bật/tắt trạng thái view-only của form dựa theo tài khoản đăng nhập
        function disableFormInputs(isDisabled) {
            const form = document.getElementById('passenger-form');
            if(!form) return;

            const inputs = form.querySelectorAll('input, textarea, button[type="submit"]');
            inputs.forEach(el => {
                el.readOnly = isDisabled;
                if (el.tagName === 'BUTTON') el.disabled = isDisabled;
            });

            // Ẩn/hiện bộ chọn giá vé nhanh
            const quickPrices = document.getElementById('form-quick-prices');
            if (quickPrices) {
                if (isDisabled) quickPrices.classList.add('hidden');
                else quickPrices.classList.remove('hidden');
            }

            // Ẩn/hiện nút đặt nhiều ghế
            const multiSelectContainer = document.getElementById('multi-select-container');
            if (multiSelectContainer) {
                if (isDisabled) multiSelectContainer.classList.add('hidden');
                else multiSelectContainer.classList.remove('hidden');
            }

            // Hiển thị cảnh báo Nhân viên chỉ được xem
            const warningBanner = document.getElementById('staff-view-only-warning');
            if (warningBanner) {
                if (isDisabled) {
                    warningBanner.classList.remove('hidden');
                    form.classList.add('opacity-85');
                } else {
                    warningBanner.classList.add('hidden');
                    form.classList.remove('opacity-85');
                }
            }
        }

        function loadSeatsToForm() {
            const tripData = state[state.currentTrip];
            const mainSeatNum = state.selectedSeats[0] || 1;
            const seat = tripData[mainSeatNum];

            renderFormSeatSelector();
            renderQuickTable();

            const formTripLabel = document.getElementById('form-trip-label');
            if (formTripLabel) formTripLabel.textContent = state.currentTrip === 'outbound' ? 'Chiều đi' : 'Chiều về';
            
            const pNameInput = document.getElementById('p-name');
            const pPhoneInput = document.getElementById('p-phone');
            const pPickupInput = document.getElementById('p-pickup');
            const pDropoffInput = document.getElementById('p-dropoff');
            const pNotesInput = document.getElementById('p-notes');
            const pPriceInput = document.getElementById('p-price');

            if(pNameInput) pNameInput.value = seat.name || '';
            if(pPhoneInput) pPhoneInput.value = seat.phone || '';
            if(pPickupInput) pPickupInput.value = seat.pickup || '';
            if(pDropoffInput) pDropoffInput.value = seat.dropoff || '';
            if(pNotesInput) pNotesInput.value = seat.notes || '';
            if(pPriceInput) pPriceInput.value = seat.price || (mainSeatNum <= 3 ? 250000 : 150000);

            // Kiểm soát quyền chỉnh sửa trên Form dựa trên phân quyền người dùng
            disableFormInputs(state.currentRole !== 'admin');

            const contactCard = document.getElementById('quick-contact-card');
            const btnCancelSeat = document.getElementById('btn-cancel-seat');

            if (seat.status === 'booked') {
                if(contactCard) contactCard.classList.remove('hidden');
                const seatNumEl = document.getElementById('contact-seat-num');
                const nameEl = document.getElementById('contact-name');
                if(seatNumEl) seatNumEl.textContent = state.selectedSeats.join(', ');
                if(nameEl) nameEl.textContent = seat.name;
                
                const cleanPhone = (seat.phone || '').trim().replace(/[^0-9+]/g, '');
                const phoneLink = document.getElementById('contact-phone-link');
                const phoneEl = document.getElementById('contact-phone');
                
                if(phoneEl) phoneEl.textContent = seat.phone || 'Chưa nhập SĐT';
                
                if (cleanPhone && phoneLink) {
                    phoneLink.setAttribute('href', `tel:${cleanPhone}`);
                } else if(phoneLink) {
                    phoneLink.setAttribute('href', '#');
                }

                // Nếu là Staff, ẩn nút Hủy Đặt Ghế của Thẻ liên lạc nhanh
                if (state.currentRole !== 'admin') {
                    btnCancelSeat?.classList.add('hidden');
                } else {
                    btnCancelSeat?.classList.remove('hidden');
                }
            } else {
                if(contactCard) contactCard.classList.add('hidden');
            }
            
            lucide.createIcons();
        }

        function selectSeat(seatNum) {
            state.selectedSeats = [seatNum];
            saveToStorage();
            loadSeatsToForm();
        }

        // Render bảng dữ liệu hành khách
        function renderQuickTable() {
            const tableBody = document.getElementById('quick-table-body');
            if(!tableBody) return;
            
            const tripData = state[state.currentTrip];
            let html = '';

            const q = state.searchQuery.toLowerCase().trim();

            for (let i = 1; i <= 6; i++) {
                const s = tripData[i];
                
                if (q !== "") {
                    const matchName = s.name.toLowerCase().includes(q);
                    const matchPhone = s.phone.toLowerCase().includes(q);
                    const matchPickup = s.pickup.toLowerCase().includes(q);
                    const matchDropoff = s.dropoff.toLowerCase().includes(q);
                    const matchNotes = s.notes.toLowerCase().includes(q);
                    const matchSeat = `ghế ${i}`.includes(q) || `ghe ${i}`.includes(q);

                    if (!matchName && !matchPhone && !matchPickup && !matchDropoff && !matchNotes && !matchSeat) {
                        continue;
                    }
                }

                const isBooked = s.status === 'booked';
                const cleanPhone = (s.phone || '').trim().replace(/[^0-9+]/g, '');
                const isSelected = state.selectedSeats.includes(i);
                
                // Nút xóa ghế chỉ hiển thị khi tài khoản có quyền Admin
                const deleteAction = (isBooked && state.currentRole === 'admin') ? `
                    <button onclick="quickClearSeat(${i})" class="p-2 text-rose-600 hover:bg-rose-50 rounded-lg transition-all" title="Xóa ghế">
                        <i data-lucide="trash-2" class="w-5 h-5"></i>
                    </button>
                ` : '';
                
                html += `
                <tr class="border-b border-slate-100 hover:bg-slate-50 transition-colors ${isSelected ? 'bg-indigo-50/50 border-l-4 border-l-indigo-600' : ''}">
                    <td class="px-4 py-4 font-bold text-slate-900">
                        <span class="inline-flex items-center justify-center w-8 h-8 rounded-full ${isBooked ? 'bg-indigo-600 text-white' : 'bg-slate-100 text-slate-500'} text-xs font-black">
                            ${i}
                        </span>
                    </td>
                    <td class="px-4 py-4">
                        ${isBooked ? `
                            <div class="font-extrabold text-slate-900 text-base">${s.name}</div>
                            <div class="mt-1.5">
                                <a href="tel:${cleanPhone}" class="inline-flex items-center gap-2 text-lg font-black text-emerald-700 hover:text-emerald-900 transition-all bg-emerald-50 hover:bg-emerald-100 active:scale-95 px-3.5 py-1.5 rounded-xl border border-emerald-200 shadow-sm">
                                    <span class="relative flex h-3 w-3">
                                      <span class="animate-ping absolute inline-flex h-full w-full rounded-full bg-emerald-400 opacity-75"></span>
                                      <span class="relative inline-flex rounded-full h-3 w-3 bg-emerald-500"></span>
                                    </span>
                                    <i data-lucide="phone" class="w-4.5 h-4.5 text-emerald-600 shrink-0"></i>
                                    <span>${s.phone}</span>
                                </a>
                            </div>
                        ` : `
                            <span class="text-slate-400 italic text-xs">Ghế trống</span>
                        `}
                    </td>
                    <td class="px-4 py-4">
                        ${isBooked ? `
                            <div class="text-xs font-semibold text-indigo-700 flex items-center gap-1">
                                <span class="inline-block w-1.5 h-1.5 rounded-full bg-indigo-500"></span> Đón: ${s.pickup || 'Chưa ghi'}
                            </div>
                            <div class="text-xs text-slate-500 flex items-center gap-1 mt-1">
                                <span class="inline-block w-1.5 h-1.5 rounded-full bg-rose-400"></span> Trả: ${s.dropoff || 'Chưa ghi'}
                            </div>
                        ` : `
                            <span class="text-slate-300">-</span>
                        `}
                    </td>
                    <td class="px-4 py-4 font-bold text-slate-800">
                        ${(s.price || 0).toLocaleString('vi-VN')} đ
                    </td>
                    <td class="px-4 py-4 text-right">
                        <div class="flex justify-end gap-1.5">
                            <button onclick="selectSeat(${i})" class="p-2 text-indigo-600 hover:bg-indigo-50 rounded-lg transition-all" title="Xem & Chọn">
                                <i data-lucide="eye" class="w-5 h-5"></i>
                            </button>
                            ${isBooked ? `
                                <button onclick="copySingleSeatInfoDirect(${i})" class="p-2 text-blue-500 hover:bg-blue-50 rounded-lg transition-all text-blue-600" title="Gửi Zalo vé này">
                                    <i data-lucide="share-2" class="w-5 h-5"></i>
                                </button>
                                ${deleteAction}
                            ` : ''}
                        </div>
                    </td>
                </tr>
                `;
            }

            if (html === '') {
                html = `
                <tr>
                    <td colspan="5" class="px-4 py-8 text-center text-slate-400 italic">
                        Không tìm thấy thông tin ghế nào phù hợp.
                    </td>
                </tr>
                `;
            }

            tableBody.innerHTML = html;
            lucide.createIcons();
        }

        // Tính doanh thu và cập nhật Dashboard hiển thị
        function updateSummaryDashboard() {
            let outboundCount = 0;
            let outboundRevenue = 0;
            
            let returnCount = 0;
            let returnRevenue = 0;

            for(let i=1; i<=6; i++) {
                const s = state.outbound[i];
                if (s.status === 'booked') {
                    outboundCount++;
                    outboundRevenue += Number(s.price || 0);
                }
            }

            for(let i=1; i<=6; i++) {
                const s = state.return[i];
                if (s.status === 'booked') {
                    returnCount++;
                    returnRevenue += Number(s.price || 0);
                }
            }

            const outSeatsEl = document.getElementById('summary-outbound-seats');
            if(outSeatsEl) outSeatsEl.textContent = `${outboundCount} / 6 ghế`;
            
            const outRevEl = document.getElementById('summary-outbound-revenue');
            if(outRevEl) outRevEl.textContent = `${outboundRevenue.toLocaleString('vi-VN')} đ`;

            const retSeatsEl = document.getElementById('summary-return-seats');
            if(retSeatsEl) retSeatsEl.textContent = `${returnCount} / 6 ghế`;

            const retRevEl = document.getElementById('summary-return-revenue');
            if(retRevEl) retRevEl.textContent = `${returnRevenue.toLocaleString('vi-VN')} đ`;

            const totalRev = outboundRevenue + returnRevenue;
            const totalRevEl = document.getElementById('summary-total-revenue');
            if(totalRevEl) totalRevEl.textContent = `${totalRev.toLocaleString('vi-VN')} đ`;
        }

        // Lưu thông tin khách hàng xuống các ghế đang chọn (Chỉ Admin)
        function savePassenger(event) {
            event.preventDefault();
            if (state.currentRole !== 'admin') {
                showToast("Tài khoản của bạn chỉ được xem danh sách, không có quyền chỉnh sửa!", "shield-alert");
                return;
            }
            
            const name = document.getElementById('p-name').value.trim();
            const phone = document.getElementById('p-phone').value.trim();
            const pickup = document.getElementById('p-pickup').value.trim();
            const dropoff = document.getElementById('p-dropoff').value.trim();
            const price = Number(document.getElementById('p-price').value || 0);
            const notes = document.getElementById('p-notes').value.trim();

            const tripData = state[state.currentTrip];

            state.selectedSeats.forEach(seatNum => {
                tripData[seatNum] = {
                    name,
                    phone,
                    pickup,
                    dropoff,
                    price, 
                    notes,
                    status: 'booked'
                };
            });

            saveToStorage();
            updateSummaryDashboard();
            
            showToast(`Đã lưu thành công thông tin cho ${state.selectedSeats.length} ghế (${state.selectedSeats.join(', ')})!`);
            
            const lastSelected = state.selectedSeats[state.selectedSeats.length - 1];
            let nextSeat = lastSelected < 6 ? lastSelected + 1 : 1;
            
            state.selectedSeats = [nextSeat];
            state.multiSelectMode = false;
            const toggleCheckbox = document.getElementById('multi-select-toggle');
            if (toggleCheckbox) toggleCheckbox.checked = false;

            saveToStorage();
            loadSeatsToForm();
        }

        // Giải phóng các ghế đang chọn về trạng thái trống (Chỉ Admin)
        function clearSelectedSeatData() {
            if (state.currentRole !== 'admin') {
                showToast("Bạn không có quyền chỉnh sửa dữ liệu!", "shield-alert");
                return;
            }

            const tripData = state[state.currentTrip];

                tripData[seatNum] = {
                    name: "",
                    phone: "",
                    pickup: "",
                    dropoff: "",
                    price: seatNum <= 3 ? 250000 : 150000,
                    notes: "",
                    status: "empty"

            loadSeatsToForm();
        }

        // Hủy nhanh một ghế cụ thể trong danh sách bảng (Chỉ Admin)
        function quickClearSeat(seatNum) {
            if (state.currentRole !== 'admin') {
                showToast("Bạn không có quyền xóa dữ liệu!", "shield-alert");
                return;
            }

            const tripData = state[state.currentTrip];
            
            tripData[seatNum] = {
                name: "",
                phone: "",
                pickup: "",
                dropoff: "",
                price: seatNum <= 3 ? 250000 : 150000,
                notes: "",
                status: "empty"
            };

            saveToStorage();
            updateSummaryDashboard();
            
            if (state.selectedSeats.includes(seatNum)) {
                if (state.selectedSeats.length > 1) {

        }

        // Sao chép thông tin vé đang chọn và tự động chia sẻ qua Zalo
        function copySelectedSeatInfo() {
            const tripData = state[state.currentTrip];
            let text = "";
            const tripName = state.currentTrip === 'outbound' ? 'CHIỀU ĐI' : 'CHIỀU VỀ';

            const bookedSeats = state.selectedSeats.filter(num => tripData[num].status === 'booked');
            
            if (bookedSeats.length === 0) {
                showToast("Các ghế đang chọn đều trống!", "alert-triangle");
                return;
            }

            if (bookedSeats.length === 1) {
                const s = tripData[bookedSeats[0]];
                text = `Ghế ${bookedSeats[0]}: ${s.name} - ${s.phone}\nĐón: ${s.pickup || '-'}\nTrả: ${s.dropoff || '-'}`;
            } else {
                const first = tripData[bookedSeats[0]];
                text = `Ghế ${bookedSeats.join(', ')}: ${first.name} - ${first.phone}\nĐón: ${first.pickup || '-'}\nTrả: ${first.dropoff || '-'}`;
            }

            if (copyTextToClipboard(text)) {
                showToast(`Đã sao chép vé và chuyển tiếp sang Zalo!`);
                redirectToZalo();
            } else {
                showToast("Không thể sao chép, vui lòng thử lại!", "alert-triangle");
            }
        }

        // Sao chép chi tiết một ghế trực tiếp trên bảng và chia sẻ qua Zalo

Đón: ${s.pickup || '-'}
Trả: ${s.dropoff || '-'}`;

            }
        }

                    text += `Trả: ${s.dropoff || '-'}\n`;

                text += `------------------------------\n`;

            text += `📊 Số khách: ${bookedCount}/6 ghế\n`;

            if (copyTextToClipboard(text)) {

                redirectToZalo();
            } else {
                showToast("Không thể sao chép, vui lòng thử lại!", "alert-triangle");
            }
        }
